
class Person {
public:
	Person() = default;
	Person(const std::string& initialName, const std::string& initialAddress) : name(initialName), address(initialAddress) {}

	std::string getName() const { return name; }
	std::string getAddress() const { return address; }

private:
	std::string name;
	std::string address;
};
